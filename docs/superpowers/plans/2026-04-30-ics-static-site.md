# ICS Static Calendar Gallery Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a pure frontend GitHub Pages site that automatically indexes root-level `.ics` files, classifies them, and presents them in a cute searchable calendar gallery with download and subscription actions.

**Architecture:** A Node build script scans root `.ics` files and writes `calendars.json`. A framework-free frontend loads that JSON and renders the gallery with safe DOM APIs instead of HTML string injection. GitHub Actions runs tests and the generator before deploying the repository as a Pages artifact.

**Tech Stack:** Static HTML, CSS, browser JavaScript, Node.js built-ins only, GitHub Actions Pages deployment.

---

## File Structure

- Create `scripts/generate-calendars.mjs`: discover `.ics` files, parse metadata, infer categories/theme visuals, and write `calendars.json`.
- Create `scripts/generate-calendars.test.mjs`: `node:test` tests for ICS parsing, classification, URL generation, and index generation.
- Create `index.html`: static page shell and accessible controls.
- Create `assets/styles.css`: cute calendar shelf visual system, responsive layout, fallback card art, list/card states, and empty/error states.
- Create `assets/app.js`: browser renderer and interaction logic using `createElement`, `textContent`, attributes, and `replaceChildren`.
- Create `assets/app.test.mjs`: `node:test` tests for pure frontend helpers exported from `assets/app.js`.
- Create `.github/workflows/pages.yml`: GitHub Pages workflow that runs tests, generates `calendars.json`, and deploys.
- Generated by script: `calendars.json`.
- Preserve existing root files: `CNAME`, `*.ics`.

```js
import { readdir, readFile, writeFile } from 'node:fs/promises';
import { fileURLToPath } from 'node:url';
import path from 'node:path';

const DEFAULT_BASE_URL = 'https://calendar.rollyourown.xyz/';

const CATEGORIES = {
  cute: { id: 'cute', label: '角色 / 萌系', emoji: '🌸', colors: ['#ffd6e7', '#fff0b8'] },
  fantasy: { id: 'fantasy', label: '影视 / 魔法', emoji: '✨', colors: ['#d9ccff', '#b8e6ff'] },
  music: { id: 'music', label: '音乐 / 演出', emoji: '🎵', colors: ['#ffd1ba', '#ffe680'] },
  birthday: { id: 'birthday', label: '生日 / 纪念日', emoji: '🎂', colors: ['#ffe2f2', '#d8f8e1'] },
  other: { id: 'other', label: '其他', emoji: '🗓️', colors: ['#dce8ff', '#f4e4ff'] },
};

export function extractCalendarTitle(fileName, icsText) {
  const calName = icsText.match(/^X-WR-CALNAME:(.+)$/im)?.[1]?.trim();
  return calName ? decodeIcsText(calName) : fileName.replace(/\.ics$/i, '').replace(/[_-]+/g, ' ').trim();
}

export function inferCategory(fileName, icsText) {
  const text = `${fileName}\n${icsText}`.toLowerCase();
  if (/chiikawa|小可爱|ちいかわ/.test(text)) return CATEGORIES.cute;
  if (/harry|potter|hp|hogwarts|魔法/.test(text)) return CATEGORIES.fantasy;
  if (/mayday|五月天|concert|演唱|music|音乐/.test(text)) return CATEGORIES.music;
  if (/birthday|birthdays|生日|anniversary|纪念/.test(text)) return CATEGORIES.birthday;
  return CATEGORIES.other;
}

export function parseIcsEvents(icsText) {
  return [...icsText.matchAll(/BEGIN:VEVENT([\s\S]*?)END:VEVENT/g)].map((match) => {
    const block = unfoldIcsLines(match[1]);
    const summary = block.match(/^SUMMARY(?:;[^:]*)?:(.+)$/im)?.[1]?.trim() ?? '未命名事件';
    const rawDate = block.match(/^DTSTART(?:;[^:]*)?:(.+)$/im)?.[1]?.trim() ?? '';
    return { summary: decodeIcsText(summary), date: normalizeIcsDate(rawDate) };
  }).filter((event) => event.date);
}

export function buildCalendarRecord(fileName, icsText, baseUrl = DEFAULT_BASE_URL) {
  const title = extractCalendarTitle(fileName, icsText);
  const category = inferCategory(fileName, icsText);
  const events = parseIcsEvents(icsText).sort((a, b) => a.date.localeCompare(b.date));
  const encodedFileName = encodeURI(fileName);
  const absoluteUrl = new URL(encodedFileName, ensureTrailingSlash(baseUrl)).toString();
  return {
    id: fileName.replace(/\.ics$/i, '').toLowerCase().replace(/[^a-z0-9一-龥]+/g, '-').replace(/^-|-$/g, ''),
    fileName,
    title,
    category: { id: category.id, label: category.label },
    url: encodedFileName,
    downloadUrl: absoluteUrl,
    webcalUrl: absoluteUrl.replace(/^https?:\/\//, 'webcal://'),
    eventCount: events.length,
    dateRange: events.length ? { start: events[0].date, end: events[events.length - 1].date } : { start: null, end: null },
    previewEvents: events.slice(0, 5),
    keywords: buildKeywords(fileName, title, events),
    visual: { emoji: category.emoji, colors: category.colors, image: pickImageCandidate(fileName, title, category.id) },
    generatedAt: new Date().toISOString(),
  };
}

export async function generateCalendarIndex({ rootDir = process.cwd(), baseUrl = DEFAULT_BASE_URL } = {}) {
  const entries = await readdir(rootDir);
  const icsFiles = entries.filter((entry) => entry.toLowerCase().endsWith('.ics')).sort((a, b) => a.localeCompare(b));
  const calendars = [];
  for (const fileName of icsFiles) {
    const filePath = path.join(rootDir, fileName);
    try {
      calendars.push(buildCalendarRecord(fileName, await readFile(filePath, 'utf8'), baseUrl));
    } catch (error) {
      const absoluteUrl = new URL(encodeURI(fileName), ensureTrailingSlash(baseUrl)).toString();
      calendars.push({
        id: fileName.replace(/\.ics$/i, ''), fileName, title: fileName.replace(/\.ics$/i, ''),
        category: { id: 'other', label: CATEGORIES.other.label }, url: encodeURI(fileName),
        downloadUrl: absoluteUrl, webcalUrl: absoluteUrl.replace(/^https?:\/\//, 'webcal://'),
        eventCount: 0, dateRange: { start: null, end: null }, previewEvents: [], keywords: [],
        visual: { emoji: CATEGORIES.other.emoji, colors: CATEGORIES.other.colors, image: null },
        parseWarning: error instanceof Error ? error.message : 'Unable to parse calendar file',
        generatedAt: new Date().toISOString(),
      });
    }
  }
  return {
    generatedAt: new Date().toISOString(),
    totalCalendars: calendars.length,
    totalEvents: calendars.reduce((sum, calendar) => sum + calendar.eventCount, 0),
    categories: Object.values(CATEGORIES).map(({ id, label, emoji }) => ({ id, label, emoji })),
    calendars,
  };
}

function unfoldIcsLines(text) { return text.replace(/\r?\n[ \t]/g, ''); }
function decodeIcsText(value) { return value.replace(/\\n/g, ' ').replace(/\\,/g, ',').replace(/\\;/g, ';').replace(/\\\\/g, '\\').trim(); }
function normalizeIcsDate(value) {
  if (/^\d{8}$/.test(value)) return `${value.slice(0, 4)}-${value.slice(4, 6)}-${value.slice(6, 8)}`;
  if (/^\d{8}T/.test(value)) return `${value.slice(0, 4)}-${value.slice(4, 6)}-${value.slice(6, 8)}`;
  return '';
}
function buildKeywords(fileName, title, events) {
  return [...new Set([fileName.replace(/\.ics$/i, ''), title, ...events.slice(0, 20).map((event) => event.summary)])].filter(Boolean);
}
function pickImageCandidate(fileName, title, categoryId) {
  const text = `${fileName} ${title}`.toLowerCase();
  if (/chiikawa|小可爱/.test(text)) return 'https://images.unsplash.com/photo-1518895949257-7621c3c786d7?auto=format&fit=crop&w=900&q=80';
  if (/mayday|五月天/.test(text)) return 'https://images.unsplash.com/photo-1493225457124-a3eb161ffa5f?auto=format&fit=crop&w=900&q=80';
  if (/harry|potter|hp/.test(text)) return 'https://images.unsplash.com/photo-1518709268805-4e9042af2176?auto=format&fit=crop&w=900&q=80';
  if (categoryId === 'birthday') return 'https://images.unsplash.com/photo-1464349095431-e9a21285b5f3?auto=format&fit=crop&w=900&q=80';
  return null;
}
function ensureTrailingSlash(value) { return value.endsWith('/') ? value : `${value}/`; }

async function main() {
  const index = await generateCalendarIndex({ rootDir: process.cwd(), baseUrl: process.env.SITE_BASE_URL ?? DEFAULT_BASE_URL });
  await writeFile(path.join(process.cwd(), 'calendars.json'), `${JSON.stringify(index, null, 2)}\n`);
}
if (process.argv[1] === fileURLToPath(import.meta.url)) main().catch((error) => { console.error(error); process.exitCode = 1; });
```

Create `assets/app.js` with:

```js
const state = { calendars: [], categories: [], query: '', category: 'all', sort: 'theme', view: 'cards', showPreview: true };

export function filterCalendars(calendars, { category, query }) {
  const normalizedQuery = query.trim().toLowerCase();
  return calendars.filter((calendar) => {
    if (category !== 'all' && calendar.category.id !== category) return false;
    if (!normalizedQuery) return true;
    const searchable = [calendar.title, calendar.category.label, ...(calendar.keywords ?? []), ...(calendar.previewEvents ?? []).map((event) => event.summary)].join(' ').toLowerCase();
    return searchable.includes(normalizedQuery);
  });
}

export function sortCalendars(calendars, sortMode) {
  const copy = [...calendars];
  if (sortMode === 'eventCount') return copy.sort((a, b) => b.eventCount - a.eventCount || a.title.localeCompare(b.title));
  if (sortMode === 'name') return copy.sort((a, b) => a.title.localeCompare(b.title));
  return copy.sort((a, b) => a.category.label.localeCompare(b.category.label) || a.title.localeCompare(b.title));
}

export function formatDateRange(dateRange) {
  if (!dateRange?.start || !dateRange?.end) return '日期待整理';
  return dateRange.start === dateRange.end ? dateRange.start : `${dateRange.start} → ${dateRange.end}`;
}

export function createSubscriptionUrl(downloadUrl) {
  return downloadUrl.replace(/^https?:\/\//, 'webcal://');
}

async function loadCalendars() {
  const response = await fetch('calendars.json', { cache: 'no-store' });
  if (!response.ok) throw new Error(`Unable to load calendar index: ${response.status}`);
  return response.json();
}

function el(tag, options = {}, children = []) {
  const node = document.createElement(tag);
  if (options.className) node.className = options.className;
  if (options.text !== undefined) node.textContent = options.text;
  if (options.hidden !== undefined) node.hidden = options.hidden;
  for (const [name, value] of Object.entries(options.attrs ?? {})) node.setAttribute(name, value);
  for (const child of children) node.append(child);
  return node;
}

function renderStats(index) {
  document.querySelector('#stat-calendars').textContent = String(index.totalCalendars ?? index.calendars.length);
  document.querySelector('#stat-events').textContent = String(index.totalEvents ?? 0);
  document.querySelector('#stat-updated').textContent = index.generatedAt ? index.generatedAt.slice(0, 10) : '—';
}

function renderCategoryFilters() {
  const container = document.querySelector('#category-filters');
  const buttons = [{ id: 'all', label: '全部', emoji: '🗂️' }, ...state.categories].map((category) => {
    const button = el('button', { className: `chip ${state.category === category.id ? 'is-active' : ''}`, attrs: { type: 'button' } }, [
      el('span', { text: category.emoji ?? '🗓️' }), document.createTextNode(category.label),
    ]);
    button.addEventListener('click', () => { state.category = category.id; render(); });
    return button;
  });
  container.replaceChildren(...buttons);
}

function render() {
  renderCategoryFilters();
  const filtered = sortCalendars(filterCalendars(state.calendars, state), state.sort);
  const grid = document.querySelector('#calendar-grid');
  grid.className = `calendar-grid ${state.view === 'list' ? 'is-list' : ''}`;
  if (filtered.length === 0) {
    grid.replaceChildren(el('div', { className: 'empty-card', text: '没有找到匹配的日历，换个关键词试试。' }));
    return;
  }
  grid.replaceChildren(...filtered.map(renderCalendarCard));
}

function renderCalendarCard(calendar) {
  const colors = calendar.visual?.colors ?? ['#dce8ff', '#f4e4ff'];
  const cover = el('div', { className: 'cover', attrs: { 'aria-hidden': 'true' } }, [el('span', { text: calendar.visual?.emoji ?? '🗓️' })]);
  cover.style.setProperty('--card-a', colors[0]);
  cover.style.setProperty('--card-b', colors[1]);
  if (calendar.visual?.image) {
    const image = el('img', { attrs: { src: calendar.visual.image, alt: '', loading: 'lazy' } });
    image.addEventListener('error', () => image.remove());
    cover.prepend(image);
  }
  const actions = el('div', { className: 'actions' }, [
    el('a', { className: 'button', text: '下载 .ics', attrs: { href: calendar.url, download: '' } }),
    el('a', { className: 'button accent', text: '订阅 webcal', attrs: { href: calendar.webcalUrl ?? createSubscriptionUrl(calendar.downloadUrl) } }),
    copyButton(calendar.downloadUrl),
  ]);
  return el('article', { className: 'calendar-card' }, [cover, el('div', { className: 'card-body' }, [
    el('div', { className: 'card-heading' }, [el('h2', { text: calendar.title }), el('span', { className: 'tag', text: calendar.category.label })]),
    el('p', { className: 'meta', text: `${calendar.eventCount} 个事件 · ${formatDateRange(calendar.dateRange)}` }),
    ...(calendar.parseWarning ? [el('p', { className: 'warning', text: `解析提示：${calendar.parseWarning}` })] : []),
    state.showPreview ? renderPreviewEvents(calendar.previewEvents ?? []) : document.createTextNode(''),
    renderDetails(calendar.downloadUrl),
    actions,
  ])]);
}

function renderPreviewEvents(events) {
  if (events.length === 0) return el('p', { className: 'preview-empty', text: '暂无事件预览' });
  return el('ul', { className: 'event-list' }, events.map((event) => el('li', {}, [el('time', { text: event.date }), el('span', { text: event.summary })])));
}

function renderDetails(downloadUrl) {
  return el('details', {}, [el('summary', { text: '本地预览 / 链接' }), el('p', { className: 'link-text', text: downloadUrl })]);
}

function copyButton(url) {
  const button = el('button', { className: 'button ghost', text: '复制链接', attrs: { type: 'button' } });
  button.addEventListener('click', async () => {
    try { await navigator.clipboard.writeText(url); button.textContent = '已复制'; }
    catch { button.textContent = url; button.classList.add('show-url'); }
  });
  return button;
}

function showStatus(message, isError = false) {
  const status = document.querySelector('#status');
  status.hidden = false;
  status.textContent = message;
  status.classList.toggle('is-error', isError);
}

function bindControls() {
  document.querySelector('#search-input').addEventListener('input', (event) => { state.query = event.target.value; render(); });
  document.querySelector('#sort-select').addEventListener('change', (event) => { state.sort = event.target.value; render(); });
  document.querySelector('#view-select').addEventListener('change', (event) => { state.view = event.target.value; render(); });
  document.querySelector('#preview-toggle').addEventListener('change', (event) => { state.showPreview = event.target.checked; render(); });
}

async function init() {
  if (!globalThis.document?.querySelector?.('#calendar-grid')) return;
  bindControls();
  try {
    const index = await loadCalendars();
    state.calendars = index.calendars ?? [];
    state.categories = index.categories ?? [];
    renderStats(index);
    render();
  } catch {
    showStatus('暂时没有读取到日历索引。请稍后刷新，或确认 calendars.json 已生成。', true);
    document.querySelector('#calendar-grid').replaceChildren(el('div', { className: 'empty-card', text: '日历展示柜正在整理中。' }));
  }
}

init();
```
